function palavraPalindromo(palavra) {
    separar = palavra.split("");
    inverter = separar.reverse();
    juntar = inverter.join("");
    if (palavra == juntar) {
        console.log(`Essa palavra "${palavra}" é um palindromo`)
    } else {
        console.log(`Essa palavra "${palavra}" NÃO é um palindromo`)
    }
}
